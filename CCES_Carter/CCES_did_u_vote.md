vote16complete <- mutate(vote16, voted = ifelse(askvote %in% 1:4,"no",
                                            ifelse(askvote %in% 5, "yes",  "NAorMissing" ))) %>%

  mutate(party = ifelse(party_ID %in% 1,"Democrat",

  # making a registered column

#vote method method

#remove old columns
